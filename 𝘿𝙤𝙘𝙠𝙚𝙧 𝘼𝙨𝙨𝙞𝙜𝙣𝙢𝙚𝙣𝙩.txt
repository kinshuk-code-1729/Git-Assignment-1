1. 𝙲𝚛𝚎𝚊𝚝𝚎 4 𝚌𝚘𝚗𝚝𝚊𝚒𝚗𝚎𝚛𝚜

(𝐍𝐨𝐭𝐞 : 𝐂𝐨𝐧𝐭𝐚𝐢𝐧𝐞𝐫𝐬 𝐜𝐚𝐧 𝐛𝐞 𝐜𝐫𝐞𝐚𝐭𝐞𝐝 𝐞𝐢𝐭𝐡𝐞𝐫 𝐛𝐲 𝐮𝐬𝐢𝐧𝐠 𝘋𝘰𝘤𝘬𝘦𝘳𝘧𝘪𝘭𝘦 𝐎𝐫 𝐛𝐲 𝐮𝐬𝐢𝐧𝐠 𝘤𝘰𝘮𝘮𝘢𝘯𝘥𝘴) 

3. 𝙿𝚞𝚜𝚑 𝚝𝚑𝚎 𝕚𝕞𝕒𝕘𝕖𝕤 𝚘𝚏 𝚊𝚕𝚕 𝚝𝚑𝚎 𝚌𝚘𝚗𝚝𝚊𝚒𝚗𝚎𝚛𝚜 𝚠𝚒𝚝𝚑 𝚜𝚙𝚎𝚌𝚒𝚏𝚒𝚎𝚍 𝚗𝚊𝚖𝚎𝚜 (𝚒.𝚎. 𝘪𝘮𝘨1, 𝘪𝘮𝘨2, 𝘪𝘮𝘨3, 𝘪𝘮𝘨4, 𝘦𝘵𝘤.) 𝚒𝚗 𝚊 ᴅᴏᴄᴋᴇʀ ʀᴇᴘᴏsɪᴛᴏʀʏ 𝚗𝚊𝚖𝚎𝚍 𝗚𝗹𝗼𝗯𝗮𝗹𝗣𝗿𝗼𝗷𝗲𝗰𝘁.
